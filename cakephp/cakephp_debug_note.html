	$this->log('test',LOG_DEBUG); // 出力先ファイル→ /app/tmp/logs/debug.log

独自のlogファイルへログ出力する

適当な箇所で下記を記述する。 

	define('BIG_CAT', 'big_cat');
	CakeLog::config('big_cat', array(
			'engine' => 'FileLog',
			'types' => array('big_cat'),
			'file' => 'big_cat',
	));

ログ出力 

	$this->log('test=neko',BIG_CAT);

自動生成された独自のlogファイル
