utils/CatService.ts

export type Cat = {
  id: number;
  name: string;
};

export class CatService {
  private cats: Cat[];

  constructor(cats: Cat[]) {
    this.cats = cats;
  }

  toUpperCaseNames(): Cat[] {
    return this.cats.map((cat) => ({
      ...cat,
      name: cat.name.toUpperCase(),
    }));
  }

  filterByName(keyword: string): Cat[] {
    return this.cats.filter((cat) => cat.name.includes(keyword));
  }
}
