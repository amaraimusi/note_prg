	/**
	 * 自然数チェック
	 * @param v チェック対処の値
	 * @returns {Boolean} true:はい  false:いいえ
	 */
	function checkNaturalNo(v) {
	
		var flg=false;
		if(v.match(/^[0-9]*$/)){
			flg=true;
		}
	
		return flg;
	
	}
