配列から重複を除去

ソースコード 

		$(function(){
			
			var ary =['abc',10,20,30,'abc',20];
			ary = array_unique(ary);

			console.log(ary);
		});
		
		/**
		* 配列から重複を除去する
		*/
		function array_unique(ary){
			var ary2= ary.filter(function (x, i, self) {
				return self.indexOf(x) === i;
			});
			
			return ary2;
		}
