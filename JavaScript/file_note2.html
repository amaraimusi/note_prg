ファイルパスからファイル名を取得する

Demo


/**
 * ファイルパスからファイル名を取得する
 * 
 * @param string fp ファイルパス
 * @return string ファイル名
 */
function _extractFileNameFromFp(fp){
	if(fp == null) return '';
	
	let ary = fp.split(/\/|\\/);
	let fn = ary[ary.length-1];
	
	return fn;
}
