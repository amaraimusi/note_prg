import { useEffect, useState } from 'react';
import {Button, Input,} from 'antd';

const Neko = ({p_nekoName, callback1}) => {

	const [nekoName, setNekoName] = useState<string>('');


	const init = () => {
		setNekoName(p_nekoName);
		console.log('リフレッシュ');//■■■□□□■■■□□□
		console.log('p_nekoName =>' + p_nekoName);//■■■□□□■■■□□□
		console.log('nekoName =>' + nekoName);// この時点でnekoNameに値はセットされていないので注意
	}


	useEffect(() => {
		init();
	}, []);


	const testFunction = () => {
		console.log(nekoName);//■■■□□□■■■□□□
		callback1(nekoName);
		
	};
  
  
	return (

			<div>
				<Button onClick={testFunction}>テスト猫ボタン</Button>
				<Input
					placeholder="Neko name"
					value={nekoName}
					onChange={(e) => setNekoName(e.target.value)}
				/>
			</div>		

	);
};

export default Neko;
